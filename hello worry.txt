local lastIndex = 0
local startTime = tick()
local typeDuration = 1.5
local typeConn

typeConn = RunService.RenderStepped:Connect(function()
	local elapsed = tick() - startTime
	local progress = math.clamp(elapsed / typeDuration, 0, 1)
	local target = math.floor(progress * totalChars)
	if target > lastIndex then
		lastIndex = target
		textLabel.Text = table.concat(chars, "", 1, target)
	end
	if progress >= 1 then
		typeConn:Disconnect()
		task.wait(0.3)
		textLabel.Text = ""
		
		local loadingLabel = Instance.new("TextLabel", bg)
		loadingLabel.Size = UDim2.new(0,300,0,22)
		loadingLabel.AnchorPoint = Vector2.new(0.5,1)
		loadingLabel.Position = UDim2.new(0.5,0,0.5,-18)
		loadingLabel.BackgroundTransparency = 1
		loadingLabel.TextColor3 = Color3.fromRGB(255,0,0)
		loadingLabel.Font = Enum.Font.Code
		loadingLabel.TextSize = 18
		loadingLabel.Text = "Loading: 0%"
		
		local container = Instance.new("Frame", bg)
		container.Size = UDim2.new(0.66,0,0,12)
		container.AnchorPoint = Vector2.new(0.5,0.5)
		container.Position = UDim2.new(0.5,0,0.5,0)
		container.BackgroundTransparency = 1
		Instance.new("UIStroke", container).Color = Color3.new(255,0,0)
		
		local fill = Instance.new("Frame", container)
		fill.Size = UDim2.new(0,0,1,0)
		fill.BackgroundColor3 = Color3.new(255,0,0)
		fill.BorderSizePixel = 0
		
		local percent = 0
		local lastUpdate = tick()
		local loadingConn
		loadingConn = RunService.RenderStepped:Connect(function()
			if percent >= 100 then
				loadingConn:Disconnect()
				gui:Destroy()
				-- Gọi kích hoạt phần 2
				if _G.StartMainHub then _G.StartMainHub() end
				return
			end
			local delta = tick() - lastUpdate
			percent = math.min(100, percent + math.ceil((delta / 1.2) * 100))
			loadingLabel.Text = "Loading: "..percent.."%"
			fill.Size = UDim2.new(percent/100, 0, 1, 0)
			lastUpdate = tick()
		end)
	end
end)

-- Hàm tạo khả năng kéo thả mượt cho Mobile
local function MakeDraggable(topbarobject, object)
	local Dragging = false
	local DragInput = nil
	local DragStart = nil
	local StartPosition = nil

	topbarobject.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			Dragging = true
			DragStart = input.Position
			StartPosition = object.Position
			
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					Dragging = false
				end
			end)
		end
	end)

	topbarobject.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			DragInput = input
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if input == DragInput and Dragging then
			local delta = input.Position - DragStart
			object.Position = UDim2.new(
				StartPosition.X.Scale, StartPosition.X.Offset + delta.X,
				StartPosition.Y.Scale, StartPosition.Y.Offset + delta.Y
			)
		end
	end)
end
