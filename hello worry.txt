local lastIndex = 0
local startTime = tick()
local typeDuration = 1.5
local typeConn

typeConn = RunService.RenderStepped:Connect(function()
	local elapsed = tick() - startTime
	local progress = math.clamp(elapsed / typeDuration, 0, 1)
	local target = math.floor(progress * totalChars)
	if target > lastIndex then
		lastIndex = target
		textLabel.Text = table.concat(chars, "", 1, target)
	end
	if progress >= 1 then
		typeConn:Disconnect()
		task.wait(0.3)
		textLabel.Text = ""
		
		local loadingLabel = Instance.new("TextLabel", bg)
		loadingLabel.Size = UDim2.new(0,300,0,22)
		loadingLabel.AnchorPoint = Vector2.new(0.5,1)
		loadingLabel.Position = UDim2.new(0.5,0,0.5,-18)
		loadingLabel.BackgroundTransparency = 1
		loadingLabel.TextColor3 = Color3.fromRGB(255,0,0)
		loadingLabel.Font = Enum.Font.Code
		loadingLabel.TextSize = 18
		loadingLabel.Text = "Loading: 0%"
		
		local container = Instance.new("Frame", bg)
		container.Size = UDim2.new(0.66,0,0,12)
		container.AnchorPoint = Vector2.new(0.5,0.5)
		container.Position = UDim2.new(0.5,0,0.5,0)
		container.BackgroundTransparency = 1
		Instance.new("UIStroke", container).Color = Color3.new(255,0,0)
		
		local fill = Instance.new("Frame", container)
		fill.Size = UDim2.new(0,0,1,0)
		fill.BackgroundColor3 = Color3.new(255,0,0)
		fill.BorderSizePixel = 0
		
		local percent = 0
		local lastUpdate = tick()
		local loadingConn
		loadingConn = RunService.RenderStepped:Connect(function()
			if percent >= 100 then
				loadingConn:Disconnect()
				gui:Destroy()
				-- Gọi kích hoạt phần 2
				if _G.StartMainHub then _G.StartMainHub() end
				return
			end
			local delta = tick() - lastUpdate
			percent = math.min(100, percent + math.ceil((delta / 1.2) * 100))
			loadingLabel.Text = "Loading: "..percent.."%"
			fill.Size = UDim2.new(percent/100, 0, 1, 0)
			lastUpdate = tick()
		end)
	end
end)

local SaveFileName = "Bom2devNgonNgu.json"
local function loadLang()
	local data = {lang = "en"}
	pcall(function()
		if readfile and isfile and isfile(SaveFileName) then
			local raw = readfile(SaveFileName)
			local ok, dec = pcall(function() return HttpService:JSONDecode(raw) end)
			if ok and type(dec) == "table" and dec.lang then data = dec end
		end
	end)
	return data
end

local allScripts = {}
local function rebuildGlobalList()
	allScripts = {}
	for _, f in ipairs(functions) do
		for _, s in ipairs(f.Scripts) do table.insert(allScripts, s) end

local createButtons, buildFuncButtons

createButtons = function(scripts)
	if not ContentRef then return end
	for _, child in ipairs(ContentRef:GetChildren()) do
		if child:IsA("TextButton") or child:IsA("UIStroke") or child:IsA("UICorner") then child:Destroy() end
	end
	local xOffset, yOffset = 0, 0
	local contentWidth = ContentRef.AbsoluteSize.X > 0 and ContentRef.AbsoluteSize.X or 310
	for _, v in ipairs(scripts) do
		local txt = getText(v.Text)
		local tSize = TextService:GetTextSize(txt, 14, Enum.Font.GothamBold, Vector2.new(math.huge, math.huge))
		local btnW = math.clamp(tSize.X + 24, 65, contentWidth - 10)
		if xOffset + btnW > contentWidth then xOffset = 0 yOffset = yOffset + 41 end
		
		local btn = Instance.new("TextButton", ContentRef)
		btn.Size = UDim2.new(0, btnW, 0, 35)
		btn.Position = UDim2.new(0, xOffset, 0, yOffset)
		btn.BackgroundColor3 = Color3.fromRGB(15, 15, 15)
		btn.Font = Enum.Font.GothamBold
		btn.TextSize = 12
		btn.Text = txt
		btn.TextColor3 = Color3.fromRGB(255, 255, 255)
		Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 8)
		Instance.new("UIStroke", btn).Color = Color3.fromRGB(255, 0, 0)
		
		btn.Activated:Connect(function()
			if v.Script then
				if type(v.Script) == "string" and loadstring then
					local c = loadstring(v.Script) if c then pcall(c) end
				end
			end
		end)
		xOffset = xOffset + btnW + 6
	end
	ContentRef.CanvasSize = UDim2.new(0, 0, 0, yOffset + 50)
end

buildFuncButtons = function()
	if not funcFrameRef then return end
	for _, child in ipairs(funcFrameRef:GetChildren()) do
		if child:IsA("TextButton") or child:IsA("UIStroke") or child:IsA("UICorner") then child:Destroy() end
	end
	for _, f in ipairs(functions) do
		local btn = Instance.new("TextButton", funcFrameRef)
		btn.Size = UDim2.new(1, -5, 0, 32)
		btn.BackgroundColor3 = Color3.fromRGB(20, 20, 20)
		btn.Font = Enum.Font.GothamBold
		btn.TextSize = 11
		btn.Text = getText(f.Text)
		btn.TextColor3 = Color3.fromRGB(255, 255, 255)
		Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 6)
		Instance.new("UIStroke", btn).Color = Color3.fromRGB(255, 0, 0)
		btn.MouseButton1Click:Connect(function()
			if searchBoxRef then searchBoxRef.Text = "" end
			createButtons(f.Scripts or {})
		end)
	end
end

_G.changeLanguage = function(lang)
	currentLanguage = lang
	pcall(function() if writefile then writefile(SaveFileName, HttpService:JSONEncode({lang = lang})) end end)
	if searchBoxRef then searchBoxRef.PlaceholderText = (lang == "vn") and "Tìm Script" or "Search" end
	rebuildGlobalList() buildFuncButtons()
	if functions[1] then createButtons(functions[1].Scripts) end
end
